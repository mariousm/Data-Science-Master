def midi_to_notes(midifile):
  midi = pm.PrettyMIDI(midifile)
  instrument = midi.instruments[0]
  notes = collections.defaultdict(list)

  # Sort the notes by start time
  sorted_notes = sorted(instrument.notes, key=lambda note: note.start)
  prev_start = sorted_notes[0].start

  for note in sorted_notes:
    start = note.start
    end = note.end
    notes['pitch'].append(note.pitch)
    notes['note'].append(pm.note_number_to_name(note.pitch))
    notes['start'].append(start)
    notes['end'].append(end)
    notes['duration'].append(end - start)
    notes['step'].append(start - prev_start)
    prev_start = start

  return pd.DataFrame({name: np.array(value) for name, value in notes.items()})

  • Qu representa cada elemento de X en la funcin de train()?

  • Justifica el uso de categorical_crossentropy como funcin de loss

En este caso estamos en un problema de clasificacin, pero no es clasificacin binaria, es decir, la variable objetivo no es 0 o 1 (por ejemplo si existe un acorde o no). Al ser un problema de clasificacin de multiclases se usa la categorical cross entropy, de lo contrario se usara la binary cross entropy.

Un detalle por si hay dudas es ver qu funcin de activacin se usa en la capa de salida, si es binaria se usa una sigmoide, pero al ser un problema de mltiples clases se usa softmax.

La diferencia entre la entrada de la red y la salida es bsicamente una, en la entrada se introducen todos los acordes menos el ltimo, y para la salida se comparan todos los acordes menos el primero.

Con la anterior lo que quiero decir es que la red neuronal busca patrones en los datos de entrada para predecir una secuencia de acordes, es decir, a partir de unos acordes cul sera el siguiente que se podra utilizar.

Esto es bsicamente lo que aprende la red, el saber qu acordes de los 50 disponibles se puedan usar para continuar con la cancin.

from scipy.spatial import distance

# Mtodo que calcula todas las distancias
def calculate_distances(pca, max=20):
  indices = [i for i in range(max)]
  pca_max = pca[:max]
  datos = np.zeros(shape=(max,max))

  # Calculamos las distancias para cada punto
  for i in range(max):
    for j in range(max):
      if i != j:
        datos[i][j] = distance.euclidean(pca_max[i], pca_max[j])

  return datos

# Mtodo que devuelve las 10 distancias mnimas
def top_10_distances(distances):
  datos = {}
  par_acordes_visitados = []

  for top in range(10):
    min_value = 100
    min_position = (-1,-1) # Indica tambin los acordes, el primer elemento de la tupla un acorde y el otro elemento el otro

    for i in range(distances.shape[0]):
      for j in range(distances[i].shape[0]):
        if i != j:
          if ((i, j) not in par_acordes_visitados):
            if distances[i][j] < min_value:
              min_value = distances[i][j]
              min_position = (i,j)

    datos[min_position] = min_value
    par_acordes_visitados.append(min_position)
    par_acordes_visitados.append((min_position[1], min_position[0])) # Guardamos el inverso, ya que la distancia es la misma de A a B que de B a A
  
  return datos
